# Base image
FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Copy requirements
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy notebook
COPY . .

# Expose the default port
EXPOSE 8888

# Start command for Voila on Render (use $PORT!)
CMD ["voila", "Untitled1.ipynb", "--port", "$PORT", "--ip", "0.0.0.0", "--no-browser"]
